    public boolean validateStackSequences(int[] pushed, int[] popped) {
        // æ€è·¯ï¼š
        // å€ŸåŠ© æ ˆï¼Œæ¨¡æ‹Ÿæƒ…æ™¯

        int len = pushed.length;
        if(len == 0) {
            return true;
        }
        LinkedList<Integer> stack = new LinkedList<>();
        int i = 0;
        int j = 0;
        while(i < len && j < len) {
            // System.out.println(pushed[i] + " : " + popped[j]);
            int exp = popped[j]; // å½“å‰éœ€è¦å¼¹å‡ºçš„èŠ‚ç‚¹å€¼
            if(!stack.isEmpty() && stack.peek() == exp) {
                stack.pop();
                // i ++;
                j ++;
            } else if(pushed[i] == exp) {
                i ++;
                j ++;
            } else {
                // æŒç»­å‹æ ˆ
                stack.push(pushed[i]);
                i ++;
            }
        }
        while(!stack.isEmpty()) {
            int val = stack.pop();
            int exp = popped[j];
            // System.out.println(val + "  " + exp);
            if(val != exp) {
                return false;
            }
            j ++;
        }
        return true;
    }
