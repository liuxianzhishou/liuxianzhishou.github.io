                                                                                                              class Solution {
    public boolean canPartition(int[] nums) {
        // жАЭиЈѓпЉЪ
        // иГМеМЕйЧЃйҐШ -- иГљеР¶е°ЂеЕЕиГМеМЕдљњеЊЧиГМеМЕдїЈеАЉжАїеТМдЄЇжАїеТМзЪДдЄАеНК
        int sum = 0;
        for(int i : nums) {
            sum += i;
        }
        if((sum & 1) == 1) {
            return false;
        } 
        // sum жШѓеБґжХ∞
        sum >>= 1;
        // зЫЃж†ЗпЉЪе∞љеПѓиГље°ЂеЕЕиГМеМЕпЉМдљњеЊЧиГМеМЕдїЈеАЉе∞љеПѓиГље§ІпЉМе¶ВжЮЬжЬАзїИиГље§Я == жАїеТМзЪДдЄАеНКпЉМиѓіжШОеПѓдї•еИЖеЙ≤з≠ЙеТМе≠РйЫЖ
        // dp[i] = dp[i - [j]] + [j]
        int len = nums.length;
        int[] dp = new int[sum + 1];
        // еЕИйБНеОЖиГМеМЕпЉМеЖНйБНеОЖзЙ©еУБпЉМдїОиАМдњЭиѓБжѓПдЄ™зЙ©еУБжЬАе§ЪеП™иГљжЛњдЄАжђ°
        for(int j = 0; j < len; j ++) {
            for(int i = sum; i - nums[j] >= 0; i --) {
                dp[i] = Math.max(dp[i], dp[i - nums[j]] + nums[j]);
            }
        }
        return dp[sum] == sum;
    }
}
