                                                                                                                                  # 解法

                                                                                                                                  class Solution {
    public int countSubstrings(String s) {
        // 思路：
        // dp[i][j] = dp[i + 1][j - 1] + 1, [i] == [j]
        int len = s.length();
        int[][] dp = new int[len][len];
        // 初始化
        int res = 0;
        // dp
        for(int i = len - 1; i >= 0; i --) {
            for(int j = i; j < len; j ++) {
                if(i == j) {
                    dp[i][j] = 1;
                    res ++;
                } else if(i == j - 1) {
                    if(s.charAt(i) == s.charAt(j)) {
                        dp[i][j] = 2;
                        res ++;
                    }
                } else {
                    // j - i >= 2
                    if(s.charAt(i) == s.charAt(j)) {
                        if(dp[i + 1][j - 1] > 0) {
                            // 说明里面的子串也是回文串
                            dp[i][j] = dp[i + 1][j - 1] + 2;
                            res ++;
                        }
                    }
                }
            }
        }
        return res;
    }
}
