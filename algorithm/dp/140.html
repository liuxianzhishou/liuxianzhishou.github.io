                                                                                                                                  class Solution {
    List<String> res;
    public List<String> wordBreak(String s, List<String> wordDict) {
        // æ€è·¯ï¼š
        // dp + å›æº¯
        // dp åˆ¤æ–­ä»¥é‚£äº›èŠ‚ç‚¹ä½œä¸ºå³è¾¹ç•Œï¼Œå¯ä»¥æ‹†åˆ†
        // å›æº¯ å¾—åˆ°æ»¡è¶³æ¡ä»¶çš„é›†åˆ
        // å¯ä»¥å°†æ»¡è¶³æ¡ä»¶çš„èµ·å§‹å•è¯ä½œä¸ºæŸ¥æ‰¾å…¥å£ï¼Œä¾æ¬¡ä»¥è¿™éƒ¨åˆ†å•è¯ä½œä¸ºå›æº¯çš„å…¥å£
        res = new ArrayList<>();
        int len = s.length();
        boolean[] dp = new boolean[len];
        List<String> list = new ArrayList<>();
        for(int i = 0; i < len; i ++) {
            for(String str: wordDict) {
                int size = str.length();
                if(i >= size - 1) {
                    String cur = s.substring(i - size + 1, i + 1); // å·¦é—­å³å¼€
                    if(cur.equals(str) && (i == size - 1 || dp[i - size])) {
                        dp[i] = true;
                        // System.out.println(i);
                        // åˆ¤æ–­æ˜¯å¦æ˜¯èµ·å§‹å•è¯
                        if(i == size - 1) {
                            list.add(cur);
                        }
                    }
                }
            }
        }
        if(!dp[len - 1]) {
            return res; // è¯´æ˜ä¸€ä¸ªæ»¡è¶³æ¡ä»¶çš„æƒ…å†µä¹Ÿä¸å­˜åœ¨
        }
        // å›æº¯
        HashSet<String> set = new HashSet<>(wordDict);
        for(String first : list) {
            LinkedList<String> path = new LinkedList<>();
            path.addLast(first);
            mySol(s, len, set, dp, first.length(), path);
        }
        return res;
    }

    private void mySol(String s,int len, HashSet<String> set, boolean[] dp, int index, LinkedList<String> path) {
        // System.out.println(index + " : " + getStr(path));
        // é€’å½’ç»ˆæ­¢æ¡ä»¶
        if(index == len) {
            res.add(getStr(path));
            return;
        }
        // index < len - 1 è¿˜éœ€è¦ç»§ç»­æ·»åŠ å•è¯
        for(int i = index; i < len; i ++) {
            if(dp[i]) {
                String tmp = s.substring(index, i + 1); // [index, i]
                if(set.contains(tmp)) {
                    // è¯´æ˜è‚¯å®šå¯ä»¥æ‹†åˆ†
                    path.addLast(tmp);
                    mySol(s, len, set, dp, i + 1, path);
                    path.removeLast();
                }
            }
        }
    }

    private String getStr(LinkedList<String> path) {
        StringBuilder sb = new StringBuilder();
        for(String cur : path) {
            sb.append(cur).append(" ");
        }
        sb.deleteCharAt(sb.length() - 1);
        return sb.toString();
    }
}
