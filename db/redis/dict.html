// åˆ¤æ–­ dict æ˜¯å¦å¤„äº é‡å“ˆå¸Œ ä¸­
#define dictIsRehashing(d) ((d)->rehashidx != -1)

/* 
 * æ¯æ¬¡è¯¥æ–¹æ³•çš„è°ƒç”¨ï¼Œè´Ÿè´£è¿ç§» n ä¸ªèŠ‚ç‚¹
 * è‹¥åœ¨è¿ç§»è¿‡ç¨‹ä¸­é‡åˆ° 10n ä¸ªèŠ‚ç‚¹ï¼Œä¹Ÿè¿”å›
 * è‹¥å…¨éƒ¨è¿ç§»å®Œæˆï¼Œè¿”å› 0ï¼›è‹¥æœ¬æ¬¡è°ƒç”¨è¿ç§»å·¥ä½œå®Œæˆï¼Œè¿”å› 1
 * */
int dictRehash(dict *d, int n) {
    // æœ€å¤šé‡åˆ° 10n ä¸ªç©ºèŠ‚ç‚¹åï¼Œç›´æ¥è¿”å›
    int empty_visits = n*10; 
    if (!dictIsRehashing(d)) return 0;

    while(n-- && d->ht_used[0] != 0) {
        dictEntry *de, *nextde;

        assert(DICTHT_SIZE(d->ht_size_exp[0]) > (unsigned long)d->rehashidx);
        // é‡åˆ° null èŠ‚ç‚¹ï¼Œç›´æ¥å¢åŠ æ­¥æ•°ï¼Œæ­¤æ—¶è‹¥ç´¯ç§¯è¾¾åˆ° 10n ä¸ªç©ºèŠ‚ç‚¹ï¼Œç›´æ¥è¿”å›
        while(d->ht_table[0][d->rehashidx] == NULL) {
            d->rehashidx++;
            // æœ¬çº¿ç¨‹å·¥ä½œè¿ç§»å®Œæˆï¼Œè¿”å› 1ï¼Œä½†æ˜¯ d->ht_used[0] != 0 è¯´æ˜è¿˜å­˜åœ¨èŠ‚ç‚¹ç­‰å¾…é‡å“ˆå¸Œ
            if (--empty_visits == 0) return 1; 
        }
        // è¯´æ˜é‡åˆ°éç©ºèŠ‚ç‚¹ï¼Œä¸”è‡ªå·±éœ€è¦è´Ÿè´£å‰ç§»è¯¥èŠ‚ç‚¹çš„å·¥ä½œ
        de = d->ht_table[0][d->rehashidx]; // å½“å‰ä¸‹æ ‡çš„å¤´èŠ‚ç‚¹
        while(de) {
            uint64_t h;

            nextde = de->next;
            // è·å–å½“å‰ key å¯¹åº”æ–°æ•°ç»„çš„ ä¸‹æ ‡
            h = dictHashKey(d, de->key) & DICTHT_SIZE_MASK(d->ht_size_exp[1]);
            // è¿ç§»
            de->next = d->ht_table[1][h];
            d->ht_table[1][h] = de;
            d->ht_used[0]--;
            d->ht_used[1]++;
            de = nextde;
        }
        d->ht_table[0][d->rehashidx] = NULL; // æ¸…ç©º
        d->rehashidx++;
    }

    /* æœ¬çº¿ç¨‹è¿ç§»å·¥ä½œå®Œæˆåï¼Œåˆ¤æ–­æ˜¯å¦å…¨éƒ¨è¿ç§»å®Œæˆï¼Œè‹¥æ˜¯ï¼Œé‡Šæ”¾ h[0] å†…å­˜ï¼Œäº¤æ¢ h[0] & h[1] å¹¶é‡ç½® h[1] */
    if (d->ht_used[0] == 0) {
        zfree(d->ht_table[0]);
        /* Copy the new ht onto the old one */
        d->ht_table[0] = d->ht_table[1];
        d->ht_used[0] = d->ht_used[1];
        d->ht_size_exp[0] = d->ht_size_exp[1];
        _dictReset(d, 1);
        d->rehashidx = -1;
        return 0;
    }

    /* è¿˜æœ‰èŠ‚ç‚¹éœ€è¦è¿ç§» */
    return 1;
}

// è‹¥ malloc_failed != null --> malloc_failed = 0 --> å…è®¸å†…å­˜åˆ†é…å¤±è´¥ï¼Œæ— è®ºå¦‚ä½•éƒ½è¦åˆ†é…å†…å­˜ç»™æ–°æ•°ç»„

// æ–°æ•°ç»„å¤§å°ï¼Œå¯ä»¥çœ‹å‡ºï¼Œå˜ä¸ºåŸæ¥çš„ 2 å€
// è´Ÿè½½å› å­ = èŠ‚ç‚¹æ•° / æ•°ç»„é•¿åº¦
// 1. è‹¥ è´Ÿè½½å› å­ >= 1 && æ²¡æœ‰åœ¨æ‰§è¡Œ bgsave | bgrewiteaofï¼Œä¼šæ‰§è¡Œ rehash
// 2. è‹¥ Redis æ­£åœ¨æ‰§è¡Œ bgsave | bgrewiteaofï¼Œåªæœ‰å½“èŠ‚ç‚¹æ•° >= 5 * æ•°ç»„é•¿åº¦æ—¶ï¼Œæ‰ä¼šè¿›è¡Œ å¼ºåˆ¶æ‰©å®¹
int _dictExpand(dict *d, unsigned long size, int* malloc_failed)
{
    if (malloc_failed) *malloc_failed = 0;

    // é‡å“ˆå¸Œä¸­ | æ•°ç»„èŠ‚ç‚¹æ•°ä¸ size å…³ç³»ä¸æ­£ç¡®æ—¶ï¼Œè¿”å› é”™è¯¯
    if (dictIsRehashing(d) || d->ht_used[0] > size)
        return DICT_ERR;

    dictEntry **new_ht_table; // å­˜æ”¾æ‰©å®¹åçš„æ•°ç»„
    unsigned long new_ht_used;
    signed char new_ht_size_exp = _dictNextExp(size);

    size_t newsize = 1ul<<new_ht_size_exp; // æ–°æ•°ç»„å¤§å°ï¼Œå¯ä»¥çœ‹å‡ºï¼Œå˜ä¸ºåŸæ¥çš„ 2 å€
    if (newsize < size || newsize * sizeof(dictEntry*) < newsize)
        return DICT_ERR;

    /* å·¦ç§»ä½æ•°å’ŒåŸæ¥ä¸€æ ·ï¼Œè¿”å› é”™è¯¯ */
    if (new_ht_size_exp == d->ht_size_exp[0]) return DICT_ERR;

    /* ç»™æ–°æ•°ç»„åˆ†é…å†…å­˜ */
    if (malloc_failed) {
        new_ht_table = ztrycalloc(newsize*sizeof(dictEntry*)); // å°è¯•åˆ†é…å†…å­˜
        *malloc_failed = new_ht_table == NULL;
        if (*malloc_failed) // å†…å­˜åˆ†é…å¤±è´¥ï¼Œè¿”å› é”™è¯¯
            return DICT_ERR;
    } else
        new_ht_table = zcalloc(newsize*sizeof(dictEntry*)); // æ— è®ºå¦‚ä½•éƒ½è¦åˆ†é…å†…å­˜

    new_ht_used = 0;

    /* Is this the first initialization? If so it's not really a rehashing
     * we just set the first hash table so that it can accept keys. */
    if (d->ht_table[0] == NULL) {
        d->ht_size_exp[0] = new_ht_size_exp;
        d->ht_used[0] = new_ht_used;
        d->ht_table[0] = new_ht_table;
        return DICT_OK;
    }

    /* Prepare a second hash table for incremental rehashing */
    d->ht_size_exp[1] = new_ht_size_exp;
    d->ht_used[1] = new_ht_used;
    d->ht_table[1] = new_ht_table;
    d->rehashidx = 0;
    return DICT_OK;
}

static dictEntry *dictGenericDelete(dict *d, const void *key, int nofree) {

    dictEntry *he, *prevHe;

    /* 1. dict ä¸ºç©ºï¼Œç›´æ¥è¿”å› null */

    /* 2. é‡å“ˆå¸Œä¸­ï¼ŒååŠ©é‡å“ˆå¸Œ */

    h = dictHashKey(d, key); // è·å– key å¯¹åº”çš„ hash å€¼

    // 3. åˆ†åˆ«åœ¨ h[0] & h[1] ä¸­æŸ¥æ‰¾

        prevHe = NULL;
